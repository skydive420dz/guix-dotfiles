Commands to capture:

#+begin_src sh
mkdir -p ~/.config/guix/machines/guixpc

{
  echo "===== uname ====="
  uname -a

  echo
  echo "===== guix describe ====="
  guix describe

  echo
  echo "===== lspci -nnk ====="
  lspci -nnk

  echo
  echo "===== lsusb ====="
  lsusb

  echo
  echo "===== lsmod ====="
  lsmod

  echo
  echo "===== network sysfs ====="
  find /sys/class/net -maxdepth 2 -type l -print

  echo
  echo "===== drm sysfs ====="
  find /sys/class/drm -maxdepth 2 -type l -print

  echo
  echo "===== firmware/kernel messages ====="
  dmesg | grep -Ei 'firmware|wifi|wlan|bluetooth|amdgpu|nvme|ahci|usb|realtek|mediatek|intel'
} > ~/.config/guix/machines/guixpc/hardware-inventory.txt
#+end_src
